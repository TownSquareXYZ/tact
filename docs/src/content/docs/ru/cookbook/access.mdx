На этой странице приведены общие примеры работы с привилегиями, владением и контролем доступа.

## Как проверить привилегии отправителя с помощью признака Ownable

```tact
// Для удобства признак Ownable необходимо импортировать из stdlib, а также из Deployable::
import "@stdlib/ownable";
import "@stdlib/deploy";

message FooBarMsg {
    newVal: Int as uint32;
}

// Ownable признак может ограничивать определенные действия только для владельца контракта SenderChecker с помощью Deployable, Ownable {
    // Постоянные состояния переменных 
    owner: Address;     // Свойство Ownable требует добавления именно этого состояния переменной 
    val: Int as uint32; // некоторое значение

    init() {
        //  мы можем инициализировать owner любым значением, которое мы хотим, в данном случае - deployer:
        self.owner = sender();
        self.val = 0;
    }

    receive("inc") {
        self.requireOwner(); // выдает код выхода 132, если отправитель не является владельцем
        self.val += 1;
    }

    receive(msg: FooBarMsg) {
        self.requireOwner(); // выдает код выхода 132, если отправитель не является владельцем
        self.val = msg.newVal;
    }
}
```
